{

Q.6 Find the Winner 

         
var Game_Winner = (S) => 
{
  let a=0;
  let  d=0;
  for(let i=0;S>i.length;i++){
    if(S[i]==='A'){
      a++;
    }
    else if(S[i]==='D'){
      d++;
    }
  }
  if(a>d){
    return "Aditya";
  }
  else if (d>a){
    return "Danish";
  }
  else
     return "Draw";
}
